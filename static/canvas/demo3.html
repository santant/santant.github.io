画布图像处理 context.drawImage:三种参数 

	  	 		window.onload = function(){
			//初始化画布
			var oCanvas = document.getElementById("canvas");
			var context = oCanvas.getContext("2d");//获得画布上下文环境
			oCanvas.width = 300;//设置画布宽高
			oCanvas.height = 300;
			var imges = new Image();
			imges.src = '../img/v2_ic01.jpg';
			imges.onload = function(){
//				context.drawImage(imges,0,0);//参数1是渲染的图片，0,0是渲染的图片从0,0坐标开始渲染
//				context.drawImage(imges,0,0,oCanvas.width,oCanvas.height);
				//参数1是渲染的图片，0,0是渲染的图片从0,0坐标开始渲染，后面2个参数是渲染之后图片的宽高,会自动缩放到那么大
				context.drawImage(imges,50,50,100,90,0,0,oCanvas.width,oCanvas.height);
				//图片对象，50,50,100,100,是原图的2个坐标，定位的1个矩形，
				//后面的四个参数0,0,oCanvas.width,oCanvas.height,就是把原图的截取的那个矩形，渲染在画布指定的4个坐标上
			}
		}

滤镜的基本算法 改变底层的rgb造成滤镜的效果 

			  window.onload = function(){	
			   var canvasa = document.getElementById("canvasa");	
			   var  contexta =  canvasa.getContext("2d");
			   var canvasb = document.getElementById("canvasb");
			   var  contextb =  canvasb.getContext("2d");
			   var image = new Image();	      
			   image.src = '../img/222.jpg';
			   image.onload = function(){
			    contexta.drawImage(image,0,0,canvasa.width,canvasa.height);
			   }
			   document.getElementById("filter").onclick = function(){		   	
			   	filter();		   				
			   }
			   function filter(){
			   	 //得到canvas的
			   	 var imageDate = contexta.getImageData(0,0,canvasa.width,canvasa.height);
	
	   			 var imgData =  imageDate.data;
	   			 for (var i = 0; i < imgData.length; i++) {
	   			 	imgData[4*i+0] = 25;
	// 			 	imgData[4*i+1] = 0;
	   			 	imgData[4*i+2] = 255;
	   			 }
				 //向画布b中put1个上面得到的画布date
				 contextb.putImageData(imageDate,0,0);//复制图像
	//		  	 contextb.putImageData(imageDate,0,0,0,0,canvasb.width,canvasb.height);
			       	
			   }
			}
