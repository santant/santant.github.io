			var oCanvas = document.getElementById("canvas");
			oCanvas.width = 800;//设置画布宽高
			oCanvas.height = 800;			
			var context = oCanvas.getContext("2d");//获得画布上下文环境
			//context绘制
			context.beginPath()//开始一段路径
			var lineearGrad = context.createLinearGradient(0,0,800,800);
			//线性渐变的变量 这里后面的坐标是控制渐变形成的区域,这里的坐标可以是负数
			//如果createLinearGradient里面的值是(0,0,800,0);那么这个渐变方向就是横向渐变的
			//如果createLinearGradient里面的值是(0,0,0,800);那么这个渐变方向就是竖着渐变的
			lineearGrad.addColorStop(0.0,'#fff');//这里的0是开始的位置
			lineearGrad.addColorStop(0.4,'#000');
			lineearGrad.addColorStop(1,'red');//1是结束的时候
			context.fillStyle = lineearGrad;//把渐变变量绘制出来
			
			context.fillRect(110,110,800,800)
			context.closePath();  			
			
			context.fill();
			
			context.stroke();
		}
