git入门详解

git 常用命令

		
	  	 		//从git上面吧东西下载到本地  －>git clone 地址
	  	 		git clone https://github.com/santant/draps.git
	  	 		
	  	 		//设置贡献者的名称和邮箱 git config --global user.name xxx

	  	 		//查看邮箱等（查询1个） git config --global xx
	  	 		git config --global user.name
	  	 		//查看git下面所有的配置 git config --global --list
	  	 		git config --global --list

三个区，工作区，缓存区（暂存区），版本区

				//查看当前区域的状态(有文件会出现状态文件)，修改之后的文件也会被显示
				git status
				
				On branch master //解释：在master分支里面
				Your branch is up-to-date with 'origin/master'.//解释:没任何提交，工作区很干净
				
				//分支有2个文件，但是还未进行版本管理
				Untracked files:
				(use "git add file..." to include in what will be committed)
				
					demo1.html
					drap.js
				
				nothing added to commit but untracked files present (use "git add" to track)//但是还未进行版本管理
				
				//吧东西添加进缓存区 //git add xx(文件名)
				git add demo1.html
				// 如果是git add . 那么就是把东西全部提交到暂存区
				
				//如果想让缓存区的东西撤销回到工作区 //git reset HEAD xxx(回到工作区的文件名)
				git reset HEAD demo1.html
				
				
				//提交到版本区  －－－－> //这里会遇到注释保存的问题 mac中按住esc 就是可以不再编辑 ，在加上：wq就说保存并退出 ：w保存，：q退出				
				git commit
				//也可以 git commit -m 后面直接跟注释
				git commit -m "这是我的注释"
				
				//直接从工作区到版本区，----->commit提交  －－－>-a就是简写的add添加到暂存区 －－－> -m注释 
				git commit －a －m "注释xx"
				
				//查看提交的历史
				git log

对比，撤销

				//对比，从工作区到暂存区
				 git diff
				 
				//对比，从暂存区到版本区 
				git diff --cached   方法2:功能相同 git diff --staged
				
				//对比，工作区到版本区 （这里的master就是你想对比分支的名字）
				git diff master
				
				//撤销，从缓存区回到工作区 -->git reset HEAD xx(文件名)
				git reset HEAD demo1.html
				
				//撤销，从工作区中的修改，撤销返回回暂存，当前的状态
				git commit －－amend
			
				//撤销，撤销一条命令，和其他文件一起提交
				git commit －－amend

git 删除和恢复

			//删除，删除暂存区里面的文件 //git rm xx（名字） 注意：这方法在工作区文件还存在的时候是无效的
			git rm demo1.js
			
			//删除，删除缓存区里面的文件，也给工作区里面的文件一起删除
			git rm －f demo1.js
			
			//删除，删除缓存区里面的文件，而工作区保留 －－> git rm --cached xx(名称)
			git rm --cached test.html
			
			//恢复(文件恢复)， git checkout （commitId）版本id 恢复文件名字
			git checkout c2d711d99326412ebd7756fa329a0f66dee3de26 drap.js 
			
			//恢复(恢复特定的版本－>根据commitId定位)  git reset --hard (commitId)版本id，恢复版本
			git reset --hard c2d711d99326412ebd7756fa329a0f66dee3de26
			
			//恢复（上一个版本）  git reset --hard HEAD^ 回到上一个版本 
			git reset --hard HEAD^
			
			//恢复好几个版本(^是上一个，～x跟多少是回退多少个版本) ～是上波浪号
			git reset --haed HEAD~2
			
			//回到现在（恢复到现在的版本） 
			通过 git reflog 
			648a0f8 HEAD@{0}: reset: moving to HEAD^
			9471f15 HEAD@{1}: reset: moving to HEAD^
			ae905e8 HEAD@{2}: reset: moving to ae905e8ec8f44a97a76ceb33afb93e6e5739a548
			9471f15 HEAD@{3}: reset: moving to 9471f15f9feb5b5b5e7e7e9d6cac3b8380d36ea5
			ae905e8 HEAD@{4}: commit (amend): zaici注销
			e311d12 HEAD@{5}: commit: disan
			9471f15 HEAD@{6}: commit: lsat
			
			//git reflog 打印出你最近操作的命令的版本，可以得到最前面的操作号id
			git reflog
			//拿到操作号id之后  git reset --hard xx(操作号id)
			git reset --hard 9471f15

多人协作，防止冲突更新代码

			//多人协作的时候，如果没同步更新的话就会常常报此错误	
			! [rejected]        master -> master (fetch first) //冲突
			error: failed to push some refs to 'https://github.com/santant/draps.git'		
				
			//从服务器更新代码（同步更新代码）
			git fetch //从远端拉取过来，自己需要手动合并
				git diff master origin/master //手动合拼之后的区别对比
				git merge origin/master  //把区别同步到自己本地（红的字是原创仓库的，正常字是本地的）
			git pull  //从远端拉取过来，自动合并

git分支问题和概念 //分支概念：如在master里面10版本的时候分支出new2版本，在版本2修改了demo1文件，提交之后，在切换到master分支，他下面的demo1文件还是以前的版本内容是没变化的

			//查看当前分支
			git branch
			
			//创建1个分支 git branch xx(创建的分支名)
			git branch new1
			
			//切换到当指定分支 git checkout xx(切换的分支名)
			git checkout new1
			
			//创建分支直接切断到创建的某个分支 git checkout -b xx(创建的分支名)
			git checkout -b new2
			//分支概念：如在master里面10版本的时候分支出new2版本，在版本2修改了demo1文件，提交之后，在切换到master分支，他下面的demo1文件还是以前的版本是没变化的
			
			//合拼分支（在master分支下）,合拼之后master就具有new2修改之后的内容了  git merge new2（分支名）
			git merge new2
			
			//查看当前分支下面合拼的分支（一般是在master分支下）
			git branch --merged    //有合并的分支才可以--merged查看
			git branch --no-merged //没合拼的分支可以用 -no-merged查看
			
			//删除某个分支（只能删除被合拼过的分支） git branch -d new2 //-d(删除) new2(分支的名字)
			git branch -d new2
			
			//删除某个分支（没有合拼的分支也会被删除），大写的D
			git branch -D new2

创建分支，同步到github上

			//吧new1这个分支提交到github上，提交到了之后到branches就会+1，如果开始只有master一个分支就是1，那么现在就是2了
			git push origin new1   //这里的分支是new1就是new1 默认是master主分支，origin是系统默认的github仓库名称，执行这句话就是在github上创建了分支，保存了你上次的东西
			
			//github上直接创建分支，直接点Branch：master（分支名），直接在输入框输出名称即可，如果要选择其他分支上建立分支，要先选择切换到其他分支在建立
			//点击github上的branches 里面可以删除和恢复分支

github上打标签（版本号）版本号可以给项目保存，不会受到其他影响

			//查看所有的历史标签，如没有为空
			git tag
			
			//给github添加1个版本//git tag xx(版本号)
			git tag v1.0
			
			//同步标签（版本信息）到github  //git push origin v1.0(版本号)
